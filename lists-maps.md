# Lists and Tuples

- list = ["videos", "of", "IT", 2020]
- hd(list)
- "videos"
- tl(list)
- ["of", "IT", 2020]

- tuple = {"videos", "of", "IT", 2020}
- elem(tuple, 2)
- "IT"

# Keyword Lists

- keyword_list = [{a: 22}, {b: 77}]
- [a: 22, b: 77]
- keyword_list[:a]

# Maps

- m = %{:a => 1, 2 => :b}
- n = %{"z" => 5, 8 => true}
- m[2]
- n["z"]
- m.a
